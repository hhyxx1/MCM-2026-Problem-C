\section*{Problem Overview}
We analyze 34 seasons of \textit{Dancing with the Stars} (DWTS) to determine whether the show's voting system—combining judges' scores with fan votes—produces fair outcomes that also maintain audience engagement.

\section*{Core Findings}

\begin{itemize}
    \item \textbf{Fan Vote Estimation:} Using Bayesian MCMC inference with Bottom-$k$ constraints, we estimated latent fan vote shares $f(i,w)$ for 2,777 contestant-week observations with \textbf{95.6\% prediction accuracy}.
    
    \item \textbf{Rank vs.\ Percentage Methods:} Rank-based aggregation produces more meritocratic outcomes (JFI = 0.727) while Percentage-based favors fans more (FFI = 0.788). Champion changed in only 3/34 seasons between methods.
    
    \item \textbf{Extreme Events:} We identified 2 instances where fan voting overrode judges in Top-3 placements (e.g., Bobby Bones S27, Bristol Palin S11).
    
    \item \textbf{Pro Dancer Effect:} Professional partners explain 38\% of judge score variance; ``Star Makers'' like Derek Hough provide +8.1 point lift.
\end{itemize}

\section*{Recommended Rule: Dynamic Log-Weighting + Judges' Save}

\begin{equation*}
\boxed{Score = \alpha(w) \cdot J\% + (1-\alpha(w)) \cdot \log(1 + F\%)}
\end{equation*}

where $\alpha(w)$ increases from \textbf{50\% to 70\%} over the season:
\begin{itemize}
    \item Weeks 1--3: 50\% judge weight (build audience engagement)
    \item Weeks 4--7: Gradual increase to 70\%
    \item Weeks 8+: 70\% judge weight (merit-focused finale)
\end{itemize}

\textbf{Judges' Save Mechanism:} When 2 contestants are in bottom, judges can save the one with higher cumulative dance scores.
